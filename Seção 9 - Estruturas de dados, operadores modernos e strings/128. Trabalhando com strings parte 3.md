## 17.2
```javascript
function getKeywordAsString(books) {
  let keywords = [];
  for(book of books) {
    keywords = [...keywords, ...book.keywords]; // Adiciona categorias
    // keywords.push(...book.keywords);
  }

  keywords = [...new Set(keywords)]; // remove duplicatas
  
  return keywords.join(';');
}
```
